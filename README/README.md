```html
<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Navbar HTML</title>

</head>
<body>

        <ul>

            <li><a href="#">Sobre</a></li>

        </ul>
    </nav>

    <div class="conteúdo">
        <!-- Conteúdo da página aqui -->
    </div>
</body>
</html>
